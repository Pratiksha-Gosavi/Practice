# Simple HPC CNN using MPI + TensorFlow

import tensorflow as tf
from mpi4py import MPI

# MPI setup
comm = MPI.COMM_WORLD
rank = comm.Get_rank()
size = comm.Get_size()

# Load MNIST dataset
(x_train, y_train), (x_test, y_test) = tf.keras.datasets.mnist.load_data()

# Normalize data
x_train = x_train / 255.0
x_test = x_test / 255.0

# Reshape for CNN
x_train = x_train[..., tf.newaxis]
x_test = x_test[..., tf.newaxis]

# Divide data among processes
n = len(x_train)
chunk = n // size
start = rank * chunk
end = (rank + 1) * chunk

x_part = x_train[start:end]
y_part = y_train[start:end]

# Simple CNN model
model = tf.keras.Sequential([
    tf.keras.layers.Conv2D(32, (3,3), activation='relu', input_shape=(28,28,1)),
    tf.keras.layers.Flatten(),
    tf.keras.layers.Dense(10, activation='softmax')
])

# Compile model
model.compile(optimizer='adam',
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])

# Train model
model.fit(x_part, y_part, epochs=3, verbose=0)

# Test accuracy
loss, acc = model.evaluate(x_test, y_test, verbose=0)

# Combine accuracy from all processes
total_acc = comm.reduce(acc, op=MPI.SUM, root=0)

# Print final accuracy
if rank == 0:
    print("Average Accuracy:", total_acc / size)
